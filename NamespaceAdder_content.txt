                log.AppendLine($"[SKIP-EXISTS] {Path.GetFileName(filePath)}");
                skipped++;

            {
                log.AppendLine($"[SKIP-HAS-NS] {Path.GetFileName(filePath)}");
                skipped++;
                continue;
            }

    private static string WrapWithNamespace(string source, string namespaceName)
    {
        var sb = new StringBuilder();

        // Split into "using block" and "rest"
        var lines = source.Split('\n');
        int lastUsing = -1;

        for (int i = 0; i < lines.Length; i++)
        {
            string trimmed = lines[i].TrimStart();
            if (trimmed.StartsWith("using ") || trimmed.StartsWith("#if ") ||
                trimmed.StartsWith("#elif ") || trimmed.StartsWith("#else") ||
                trimmed.StartsWith("#endif") || trimmed.StartsWith("//") ||
                string.IsNullOrWhiteSpace(trimmed))
            {
                // continue scanning
            }
            else
            {
                // First non-using, non-comment, non-empty line
                lastUsing = i;
                break;
            }
        }

        // Collect header (using lines + blank lines before first type)
        var headerLines = new System.Collections.Generic.List<string>();
        var bodyLines   = new System.Collections.Generic.List<string>();

        for (int i = 0; i < lines.Length; i++)
        {
            if (i < lastUsing)
                headerLines.Add(lines[i]);
            else
                bodyLines.Add(lines[i]);
        }

        // Write header as-is
        if (headerLines.Count > 0)
        {
            sb.Append(string.Join("\n", headerLines));
            sb.Append("\n");
        }

        // Open namespace
        sb.AppendLine($"\nnamespace {namespaceName}");
        sb.AppendLine("{");

        // Indent body by 4 spaces
        foreach (var line in bodyLines)
        {
            if (string.IsNullOrWhiteSpace(line))
                sb.AppendLine();
            else
                sb.AppendLine("    " + line.TrimEnd('\r'));
        }

        // Close namespace
        sb.AppendLine("}");

        return sb.ToString();
    }
